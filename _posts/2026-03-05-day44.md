#### 함수의 매개변수, 반환값 힌트 적용하기
함수는 변수나 반환값의 타입을 명시하지 않으므로 개발자가 타입혼동을 할 수 있기 때문에 힌트를 작성할 수 있습니다.

```
def func01(arg1, arg2): # 정수 더하기 연산 함수
    return arg1+arg2

result = func01(1,3)
print(result) # 4

result = func01('a','b')
print(result) # ab


# 타입 힌트를 갖는 함수 정의
def func02(arg1:int, arg2:int)->int:
    return arg1+arg2

result = func02(1,3)
print(result) # 4

result = func02('a','b') # 함수에 선언된 타입은 단순 힌트일 뿐 실행에는 영향을 주지 않습니다.
print(result) # ab 
```


## 파일 읽기 쓰기

```
# 읽기

# 파일불러오기
f = open('C:/dev/workspace/python/workspace/03.05/data.txt','r', encoding='utf-8') # 해당 경로 파일을 읽기 모드의 인코딩 utf-8로 불러오기

# 여러줄을 반복 객체로 읽기
lines = f.readlines()

for line in lines:
    print(line)
    
f.close() # 사용한 파일 닫기

print('-'*40)

# 한줄 한줄 읽어보기(이터레이터 형식으로)
f = open('workspace/03.05/data.txt','r', encoding='utf-8')

while True:
    # 한줄 한줄 읽어오다가 더 이상 읽을 행이 없으면 중지
    line = f.readline() # 더 이상 읽을 행이 없으면 '' 반환
    
    if not line:
        break
    else:
        print(line)
        
f.close()

print('-'*40)

# 한번에 모든 텍스 텍스트 파일 읽어오기
f = open('workspace/03.05/data.txt','r', encoding='utf-8')

data = f.read()
print(data)

f.close()
# 한줄씩 읽어올 때는 \n(줄바꿈)도 한 라인으로 간주하여 읽지만 한번에 모든 텍스트를 읽어올때는 해당 x
```

#### 컨텍스트 매니저 문법
리소스를 사용한 후 자동으로 닫아주는 문법입니다.

<pre>
with open(경로, 모드) as 파일객체변수명:
    파일객체변수.write() ...

* with문의 들여쓰기 구간이 종료되면 리소스는 자동 close() 호출합니다.
</pre>

```
#f = open('workspace/03.05/data.txt','r', encoding='utf-8')
with open('workspace/03.05/data.txt','r', encoding='utf-8') as f:
    lines = f.readlines()
    for line in lines:
        print(line)

with open('workspace/03.05/data2.txt','r', encoding='utf-8') as f:
    while True:
        line = f.readline()
        if not line:
            break
        print(line)
```


## 클래스와 객체
- 파이썬은 객체지향이기도 하기 때문에 객체를 생성할 수 있는 설계도 클래스를 정의할 수 있습니다.

```
class Sample:

```

```

    pass

class Calculator:
    # 사칙연산 메소드 정의하기
    
    def add(self,a,b): # 파이썬에서 메소드의 매개변수 1번은 무조건 자신 객체를 받는 변수입니다. 
        return a+b
    
    def minus(self,a,b):
        return a-b
    
    def multi(self,*args):
        if len(args)>0:
            total = 1
            for element in args:
                total *= element
            return total
        else:
            return None
    
c1 = Calculator()
print(c1.add(1,2)) # 호출 인수는 첫번째 값이 두번째 매개변수로 전달됩니다.
print(c1.minus(5,1))

result = c1.multi(1,2,3,4,5)
print(result)


class Sample2:
    field1 = '속성1' # 정적 필드 선언
    
s2_1 = Sample2()

print(s2_1.field1)
print(s2_2.field1)
print(Sample2.field1) # 정적필드 사용은 클래스, 객체 전부 사용가능

print('='*10)

s2_1.field1 = '변경 필드' # 객체를 통하여 정적 필드 값을 수정하는 순간 인스턴스 필드로 변경합니다.
print(s2_1.field1)
print(s2_2.field1)
print(Sample2.field1)

print('='*10)

Sample2.field1 = '클래스로 변경' # 클래스를 통하여 정적 필드 값을 수정하면 모든 객체의 정적 필드 값 적용 (단, 인스턴스 필드로 수정된 객체는 제외)
print(s2_1.field1)
print(s2_2.field1)
print(Sample2.field1)
```
