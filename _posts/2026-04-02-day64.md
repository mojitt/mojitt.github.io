## Form
#### 폼이란?
사용자로부터 입력을 받기 위해 사용하는 양식

#### 제어 컴포넌트
- 사용자가 입력한 값에 접근하고 제어할 수 있게 해주는 컴포넌트
- 값이 리액트의 통제를 받는 입력 폼 엘리먼트


#### 여러 개의 입력 다루기
컴포넌트에 여러 개의 State를 선언하여 각각의 입력에 대해 사용하면 됨

#### input Null Value
- value prop은 넣되 자유롭게 입력할 수 있게 만들고 싶을 경우, 값에 underfined 또는 null을 넣으면 됨

```
// State 사용

import { useState } from "react"

const FormCompo = () => {
    const [text, setText] = useState("안녕하세요.")
    const [email, setEmail] = useState("example@test.com")
    const [textArea, setTextArea] = useState("내용을 입력하세요.")
    const [select, setSelect] = useState('item2')
    const [radio, setRadio] = useState('n')
    const [check, setCheck] = useState(['item2', 'item3'])

    const handlerCheckBox = (e) => {
        const {value, checked} = e.target
        if(checked){ // 선택했다. 즉 체크하겠다.
            setCheck([...check,value])
        }else{ // 선택해제하겠다.
            setCheck(check.filter((item)=> item !== value))
        }
    }
    
    return (
        <form>
            <input type="text" value={text} onChange={(e)=>{setText(e.target.value)}}/> <br />
            <input type="text" defaultValue={"안녕하세요."}/> <br />
            <input type="email" value={email} onChange={(e)=>{setEmail(e.target.value)}}/>  <br />
            {/* textarea 태그도 value 속성으로 초기값 제어를 합니다. */}
            <textarea value={textArea} onChange={(e)=>{setTextArea(e.target.value)}}></textarea> <br />
            <select value={select} onChange={(e)=>{setSelect(e.target.value)}}>
                <option value="item1">항목1</option>
                <option value="item2">항목2</option>
                <option value="item3">항목3</option>
            </select> <br />
            <input type="radio" name="isAgree" value='y'
                    onChange={(e)=>{setRadio(e.target.value)}} checked={radio=='y'&&true}/>동의
            <input type="radio" name="isAgree" value='n' 
                    onChange={(e)=>{setRadio(e.target.value)}} checked={radio=='n'&&true}/>비동의 <br />
            <input type="checkbox" name="checkGroup" value="item1" onChange={handlerCheckBox} />항목1
            <input type="checkbox" name="checkGroup" value="item2" onChange={handlerCheckBox} checked={check.includes('item2')}/>항목2
            <input type="checkbox" name="checkGroup" value="item3" onChange={handlerCheckBox} checked={check.includes('item3')}/>항목3
        </form>
    )
}

export default FormCompo
```

```
// useForm 사용

import { useForm } from 'react-hook-form'

const UseFormCompo = () => {

    /*
     *useForm이 제공하는 옵션들*
     register : 양식 등록을 위한 함수
     handleSubmit : 제출버튼 클릭시 이벤트 제어 함수
     formState:{errors} : 유효성 검사 후 에러메시지를 갖는 객체 (errors.이름.message)
     reset : 초기화 함수
     watch : 양식 값들 실시간 확인 (watch().이름)
     
     *useForm에 제공하는 옵션들*
     mods : 유효성 검사 이벤트 등록 (onSubmit, onBlur, onChange, all(기본))
     defaultValues : 입력양식 초기값 담는 객체
     */
    const {
        register,
        handleSubmit,
        formState:{errors},
        reset,
        watch,
    } = useForm({
        mode : "onSubmit", /* onSubmit, onBlur, onChange, all(기본) */
        reValidateMode:"onSubmit", /* 유효성 검사 최초 실패시 이후 유효성 검사 모드 onSubmit, onBlur, onChange */
        defaultValues : {
            tx1 : '초기값',
            group1 : 'btn2',
            group2 : ['check1','check3'],
            select1 : 'item2',
        }
    })

    const onSubmit = (data) => {
        console.log(data)
        alert('전송완료!')
        reset()
    }

    return (
        <form onSubmit={handleSubmit(onSubmit)}>
            <div>
                text : <input type="text" {...register("tx1",{required:'tx1은 필수 입력입니다.'})}/> <br />
                <div style={{color:"red"}}>{errors.tx1 && errors.tx1.message}</div>
            </div>
            <div>
                {/* 아래 입력양식의 이름은 tx2로 짓고 전송버튼 클릭시 필수 입력요소가 될 수 있도록 useForm 등록해보세요. */}
                text2 : <input type="text" {...register("tx2",{required:'tx2은 필수 입력입니다.', maxLength:{value:5,message:'최대길이는 5자리입니다.'}})}/>
                <div style={{color:"red"}}>{errors.tx2 && errors.tx2.message}</div>{/* 예외 메시지 출력 영역 */}
            </div>
            <button>전송</button>
            <div>
                {watch().tx1}
            </div>
            <div>
                radio :
                <input type="radio" name="group1" value="btn1" {...register("group1")}/>버튼1
                <input type="radio" name="group1" value="btn2" {...register("group1")}/>버튼2
            </div>
            <div>
                checkbox :
                <input type="checkbox" name="group2" value="check1" {...register("group2")}/>체크1
                <input type="checkbox" name="group2" value="check2" {...register("group2")}/>체크2
                <input type="checkbox" name="group2" value="check3" {...register("group2")}/>체크3
            </div>
            <div>
                select :
                <select {...register("select1")}>
                    <option value="item1">항목1</option>
                    <option value="item2">항목2</option>
                    <option value="item3">항목3</option>
                </select>
            </div>
        </form>
    )
}
export default UseFormCompo
```
